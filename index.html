## We are in the pipe, five by five.

import cv2
import os

imgfile_list = os.listdir()

set_height = 720
set_width = 2560
target_width = int(set_width/2)

for item in imgfile_list:
    try:
        img = cv2.imread(item)
        height = img.shape[0]
        width = img.shape[1] 
        #print(height,width)
        
        if width == set_width:
            cropped = img[0:set_height, target_width:set_width]  # 裁剪坐标为[y0:y1, x0:x1]
            cv2.imwrite('right_' + item , cropped)  
            cropped = img[0:set_height, 0:target_width]  # 裁剪坐标为[y0:y1, x0:x1]
            cv2.imwrite('left_' + item , cropped)  
    except:
        pass
